  • 弹出普通文本信息提示以及位置演示：

    x.dom("#id").on('click',function(){
    x.ui.dialogs.newDialog(); //默认
    x.ui.dialogs.newDialog({offsets:"left-top"}); //左上角
    x.ui.dialogs.newDialog({offsets:"right-top"}); //右上角
    x.ui.dialogs.newDialog({offsets:"left-bottom"}); //左下角
    x.ui.dialogs.newDialog({offsets:"right-bottom"}); //右下角
    x.ui.dialogs.newDialog({offsets:"left-middle"}); //靠左居中
    x.ui.dialogs.newDialog({offsets:"right-middle"}); //靠右居中
    x.ui.dialogs.newDialog({offsets:"middle-top"}); //居中置顶
    x.ui.dialogs.newDialog({offsets:"middle-bottom"}); //居中置底
    x.ui.dialogs.newDialog({offsets:{left:"100px",top:"100px"}}); //自定义
});
